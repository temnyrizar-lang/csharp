Сегодня изучил цикл while и сделал простенькую проверку пароля:

```csharp
using System;
class Program
{
    static void Main()
    {
        string password = "swinge-des-todes";
        int count = 0;
        Console.Write("Введите пароль: ");
        string userpass = Console.ReadLine();
        while (true)
        {
            count++;
            if (userpass != password && count >= 8)
            {
                Console.WriteLine("Слишком много попыток (8). Повторите позже.");
                break;
            }
            else if (userpass != password && count < 8)
            {
                Console.Write("Повторите ещё раз: ");
                userpass = Console.ReadLine();
            }
            else
            {
                Console.WriteLine("Доступ разрешён!");
                break;
            }
        }
    }
}
```
